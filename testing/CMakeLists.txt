## ----------------------------------------------------------------------------
## 1. TestUtils helper library
##
add_library(TestUtils STATIC
  TestUtils/G4HepEmDataComparison.hh
  TestUtils/G4SetUp.hh
  TestUtils/G4SetUp.cc
  TestUtils/Hist.hh)
